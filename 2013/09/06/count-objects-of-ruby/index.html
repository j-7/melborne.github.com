+ module ObjectTrapper
+   def new(*args, &blk)
+     obj = allocate
+     obj.send(:initialize, *args, &blk)
+     objects << obj
+     obj
+   end
+   
+   def objects
+     @objects ||= []
+   end
+ end

Stock = Struct.new(:name, :market, :ticker, :sector, :website, :desc) do
+ extend ObjectTrapper
  
  def to_s
    format = <<-EOS
    %s (%s:%s)
      sec: %s
      url: %s
      des: %s
    ----------
    EOS
    format % values
  end
end
